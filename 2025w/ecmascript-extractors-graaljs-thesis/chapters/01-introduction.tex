class Point {
    #x;
    #y;
    constructor(x, y) {
        this.#x = x;
        this.#y = y;
    }

    static [Symbol.customMatcher](subject) {
        return #x in subject ? [subject.#x, subject.#y] : false;
    }
}

const p = new Point(1, 2); // Construction
const Point(x, y) = p; // Extraction (calls Point[Symbol.customMatcher](p))
x; // 1
y; // 2
